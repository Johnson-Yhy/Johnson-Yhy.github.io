#include<iostream>

using namespace std;

#define InitSize 10

typedef struct{
    int *data;
    int length;
    int MaxSize;
}SqList;

void InitList(SqList &L){
    L.data=(int *)malloc(InitSize*sizeof(int));
    L.length=0;
    L.MaxSize=InitSize;
}

bool deleteList(SqList &L,int &s,int &t){
    int j=0;
    if(L.length==0||s>=t)
    return false;
    else{
        for(int i=0;i<L.length;i++){
            if(L.data[i]<s||L.data[i]>t){
                L.data[j]=L.data[i];
                j++;
            }
        }
        L.length=j;
        return true;
    }
}

void PrintList(SqList &L){
    for(int i=0;i<L.length;i++){
        cout<<L.data[i]<<" ";
    }
    cout<<endl;
}

int main(){
    SqList L;
    InitList(L);
    cin>>L.length;
    for(int i=0;i<L.length;i++){
        cin>>L.data[i];
    }
    int s,t;
    cin>>s>>t;
    deleteList(L,s,t);
    PrintList(L);
    return 0;
}
